google-genai          # Gemini SDK: generation + embeddings + token counting
chromadb              # local persistent vector store (cosine top-k search)

pymupdf               # fallback PDF extraction (handles trickier layouts)
fastapi               # API server
uvicorn[standard]     # ASGI server that runs FastAPI (with websockets/SSE extras)
pydantic              # request/response validation
python-multipart      # parse multipart/form-data for file uploads (POST /ingest)
python-dotenv         # load GEMINI_API_KEY from .env

torch                 # backs the cross-encoder — install the CPU build (see note above)
